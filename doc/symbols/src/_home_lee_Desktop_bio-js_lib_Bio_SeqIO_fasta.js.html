1 /**
  2  * @lends Bio.SeqIO.fasta
  3  */
  4 Bio.functions.include_once("Bio::Seq::primaryseq");
  5 Bio.SeqIO.fasta = Class.create(Bio.SeqIO,{
  6   /**
  7    * Bio.SeqIO.fasta
  8    * @author Lee Katz <lskatz@gmail.com>
  9    * @class Input/output stream for fasta files
 10    * @constructs
 11    * @extends Bio.SeqIO
 12    */
 13   initialize: function($super,el,args){
 14     $super(el,args);
 15     
 16     // if the mode is write, then set the value back to ""
 17     if(this.options.mode.match(/w/) && !this.options.mode.match(/\+/)){
 18       this.element.update();
 19     }
 20     //    TODO if the mode is append, then do not set the value back to ""
 21     
 22     var el=this.element;
 23     var value=el.innerHTML;
 24     if(el.value) value=el.value;
 25     if(value) this._parseFile(value);
 26   },
 27   /**
 28    * Gives a Bio.Seq object or false if there are no more sequences
 29    * @returns Bio.Seq
 30    */
 31   next_seq:function(){
 32     if(this._seqCounter>=this.seq.length){
 33       this.reset();
 34       return false;
 35     }
 36     var seq=this.seq[this._seqCounter];
 37     this._seqCounter++;
 38     return seq;
 39   },
 40   /**
 41    * Writes a single sequence to the document
 42    * @todo Be able to tie the function to an element's value or innerHTML
 43    * @param Seq A Seq object
 44    * @returns void
 45    */
 46   write_seq:function(Seq){
 47     var sequence=Seq.seq().replace(/(.{60})/g,"$1"+this.nl);
 48     var id=Seq.id();
 49     var desc=Seq.desc();
 50     var nl=this.nl;
 51     if(this.options.html) nl="<br />"+nl;
 52     var fasta=">"+id+" "+desc+nl+sequence+nl;
 53     
 54     if(this.element){
 55       this.element.insert(fasta);
 56     } else {
 57       document.write(fasta);
 58     }
 59   },
 60   /**
 61    * reset the iterator for the Seq array
 62    * @returns void
 63    */
 64   reset:function($super){
 65     this._seqCounter=0;
 66   },
 67   /**
 68    * Parses a fasta string for sequences and returns an array of sequences
 69    * @returns Array of Seqs
 70    * @private
 71    */
 72   _parseFile:function(str){
 73     if(this.options.html){
 74       str=str.replace(/<[^>]+?>/g,""); // strip tags
 75       str=str.replace(/>/g,">");
 76     }
 77     var seqArray=[];
 78     var seqEntry=str.split(/>/);
 79     seqEntry=seqEntry.without("");
 80     for(var i=0;i<seqEntry.length;i++){
 81       seqEntry[i]=seqEntry[i].replace(/^\s+|\s+$/g,""); // trim
 82       if(seqEntry[i]=="") continue;
 83       var seq=this._parseSeqEntry(seqEntry[i]);
 84       seqArray.push(seq);
 85     }
 86     //seqArray=seqArray.grep(/>/);
 87     this.seq=seqArray;
 88     this.reset();
 89     return seqArray;
 90   },
 91   /**
 92    * Parses one sequence entry
 93    * @private
 94    * @returns Seq
 95    */
 96   _parseSeqEntry:function(entry){
 97     //var line=entry.split(/\n/);
 98     var line=entry.split(this.nl);
 99     var defline=line.shift();
100     var id=defline.replace(/\s+.+$/,"");
101     var desc=defline.replace(/^\S+\s+/,"");
102     
103     var sequence=line.join("").replace(/\s+/g,"");
104     
105     var seq=new Bio.Seq.primaryseq({id:id,desc:desc,seq:sequence});
106     return seq;
107   },
108 	toString:function(){
109 	  var returnStr="";
110 	  for(i=0;i<this.seq.length;i++){
111 	    returnStr+=this.seq[i]+this.nl;
112 	  }
113 	  return returnStr;
114 	}
115 });
