 39     for(var i=0;i<line.length;i++){
 40       if(line[i].match(/^>/)){
 41         i++;
 42         // if it has something other than ATCG or whitespace then it is not a fasta
 43         if(line[i].match(/[^ATCG\s]/)){
 44           i--;
 45           continue;
 46         }
 47         return 'fasta';
 48       }
 49     }
