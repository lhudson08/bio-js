1 Bio.functions.include_once("Bio::Align");
  2 /**
  3  * @lends Bio.AlignIO
  4  */
  5 Bio.AlignIO = Class.create(Bio.Root,{ 
  6   /**
  7    * Bio.AlignIO
  8    * @author Lee Katz <lskatz@gmail.com>
  9    * @class Input/output stream for multiple sequence alignments
 10    * @constructs
 11    * @extends Bio.Root
 12    * @todo somehow use $super to get default variables too
 13    */
 14   initialize: function($super) {
 15     this.options = Object.extend({
 16       file:false, // TODO accept a filename
 17       format:false,
 18       fileText:false // the string contents of a file
 19     }, arguments[1]|| { });
 20     // check for required parameters
 21     if(this.options.fileText==false){
 22       this.throw("AlignIO requires a fileText parameter");
 23     }
 24     
 25     // get the correct AlignIO object
 26     if(!this.options.format){
 27       this.options.format=this.guessFormat(this.options.fileText);
 28     }
 29     var format=this.options.format;
 30     return new Bio.AlignIO[format](this.options);
 31   },
 32   /** 
 33    * returns a string representing the format
 34    * @param str The sequence string
 35    * @TODO do more than just FASTA
 36    * @returns String
 37    */
 38   guessFormat:function(str){
 39     var line=str.split("\n");
 40     for(var i=0;i<line.length;i++){
 41       if(line[i].match(/^>/)){
 42         i++;
 43         // if it has something other than ATCG or whitespace then it is not a fasta
 44         if(line[i].match(/[^ATCG\s]/)){
 45           i--;
 46           continue;
 47         }
 48         return 'fasta';
 49       }
 50     }
 51     this.throw("Could not determine the type of sequence file");
 52   },
 53   /**
 54    * Gets the next sequence and returns a Bio::Align compatible object
 55    * @returns Bio.Align
 56    */
 57   next_aln:function(){
 58 		this.notImplemented();
 59 	}
 60 });
